= libconfig

== installation

[source, sh]
----
git clone git@github.com:hyperrealm/libconfig.git
cd libconfig
autoreconf
./configure --prefix=${HOME}/prefix/libconfig
make -j
make install
----

== pkgconfig

add path for pkg config

[source, sh]
----
export PKG_CONFIG_PATH=${PKG_CONFIG_PATH}:${HOME}/prefix/libconfig/lib/pkgconfig
----
