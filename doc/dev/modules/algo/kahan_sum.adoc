Example
[source,cpp]
----
kahan_sum *k;
kahan_sum_ini(&k);
for (i = 0; i < n; i ++)
    kahan_sum_add(kahan_sum, a[i]);
sum = kahan_sum_get(kahan_sum);
kahan_sum_fin(k);
